c++

#include <iostream>
#include <string>
using namespace std;
class Solution {
public:
    static bool isPalindrome(int x) {
        string original = to_string(x);
        string reversed = to_string(x);
        reverse(reversed.begin(), reversed.end());
        return (original == reversed) ? true : false;
    }
};
int main() {
    cout << Solution::isPalindrome(121) << endl;
    cout << Solution::isPalindrome(123) << endl;
    return 0;
}
